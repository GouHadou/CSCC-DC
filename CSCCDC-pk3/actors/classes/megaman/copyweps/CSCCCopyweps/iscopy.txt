actor IceBladeWep : BaseMM8BDMWep 
{
Weapon.AmmoUse 3
Weapon.AmmoGive 28
Weapon.SlotNumber 2
Obituary "$OB_ICEBLADEBOSS"
Inventory.Pickupmessage "$PU_ICEBLADEBOSS"
weapon.ammotype "IceBladeAmmo"
inventory.icon "ICBUI"
Tag "$TAG_ICEBLADE"
States
{
Spawn:
ICBU A 1
loop
Ready:
ICBU B 0 ACS_ExecuteAlways(998,0,DYE_ICE) 
ICBU B 1 A_WeaponReady
Goto Ready+1
Deselect:
TNT1 AAAAAAAAAAAAAAAAAAAAAA 0 A_Lower
ICBU B 1 A_Lower
Loop
Select:
TNT1 AAAAAAAAAAAAAAAAAAAAAA 0 A_Raise
ICBU B 1 A_Raise
Loop
Fire:
ICBU B 0 A_JumpIfNoAmmo("NoAmmo")
ICBU B 0 A_PlaySoundEx("weapon/mbuster","Weapon")
ICBU B 0 A_FireCustomMissile("IceBlade2",0,1,8,0)
ICBU BCD 2
ICBU B 10
ICBU B 0 A_JumpIfInventory("IsBot", 1, "BotCheck")
Goto Hold
Hold:
ICBU B 1 A_RailWait
ICBU B 0 A_Refire
ICBU B 10 A_GiveInventory("IceBladeFlag",2)
ICBU B 0 A_TakeInventory("IceBladeFlag",999)
Goto Ready+1
BotCheck:
ICBU B 5 A_RailWait
ICBU B 0 A_Jump(128, "BotCheck")
Goto Hold+2
NoAmmo:
ICBU B 1 ACS_Execute(979,0)
Goto Hold
}
}

{
PROJECTILE
Radius 15
Height 8
scale 2.5
damage (20)
Obituary "$OB_ICEBLADEBOSS"
damagetype "IceBlade"
speed 40
States
{
Spawn:
TNT1 A 0
TNT1 A 0 A_JumpIfInTargetInventory("LightTeamFlag",1,"BFire")
TNT1 A 0 A_JumpIfInTargetInventory("WilyTeamFlag",1,"RFire")
TNT1 A 0 A_JumpIfInTargetInventory("CossackTeamFlag",1,"GFire")
TNT1 A 0 A_JumpIfInTargetInventory("KingTeamFlag",1,"YFire")
Goto LoopJump
BFire:
TNT1 A 0 Thing_SetTranslation(0, 230)
Goto LoopJump
RFire:
TNT1 A 0 Thing_SetTranslation(0, 231)
Goto LoopJump
GFire:
TNT1 A 0 Thing_SetTranslation(0, 232)
Goto LoopJump
YFire:
TNT1 A 0 Thing_SetTranslation(0, 233)
Goto LoopJump
LoopJump:
TNT1 A 0 A_Jump(256, "Spawn2")
Goto Spawn2
Spawn2:
ICBL A 0
ICBL A 0 A_PlaySoundEx("weapon/centaurflash","Weapon")
ICBL AA 1 A_JumpIfInTargetInventory("IceBladeFlag",2,"Split")
ICBL B 0 A_SpawnItemEx("IceBladeFX", -8, Random(-16, 16), random(0, -8))
ICBL BB 1 A_JumpIfInTargetInventory("IceBladeFlag",2,"Split")
loop
Death:
Goto Split //Experimenting
XDeath:
TNT1 A 0 
stop
Split:
MMFX B 0 A_SpawnItemEx("IceBit",5,0,0,Random(5, 10),0,10,60)
MMFX B 0 A_SpawnItemEx("IceBit",5,0,0,Random(5, 10),0,10,-60)
MMFX B 0 A_SpawnItemEx("IceBit",5,0,0,Random(5, 10),0,10,120)
MMFX B 0 A_SpawnItemEx("IceBit",5,0,0,Random(5, 10),0,10,-120)
MMFX B 0 A_SpawnItemEx("IceBit",5,0,0,Random(5, 10),0,10,180)
MMFX B 1 A_SpawnItemEx("IceBit",5,0,0,Random(5, 10),0,10,0)
stop
}
}
