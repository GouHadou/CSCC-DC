actor HydroCannonWep : CSCCWepBase
{
Weapon.AmmoUse 1
Weapon.AmmoGive 28
Weapon.SlotNumber 7
Obituary "%o was hosed down by %k's Hydro Cannon."
weapon.ammotype "HydroCannonAmmo"
inventory.icon "HYDROI"
DropItem "WaterBalloonWepCS"
States
{
Spawn:
TNT1 A 1 
stop
Ready:
HCAN B 0 ACS_ExecuteAlways(998,0,DYE_GIZMO)
Ready2:
HCAN B 0 A_JumpIfNoAmmo("Ready3")
HCAN B 7 A_WeaponReady
HCAN B 0 A_GiveInventory("GizmoAllGiver",1)
Goto Ready2
Ready3:
HCAN B 7 A_WeaponReady(WRF_NOFIRE)
HCAN B 0 A_GiveInventory("GizmoAllGiver",1)
HCAN B 0 A_JumpIfNoAmmo("Ready3")
Goto Ready2
Deselect:
TNT1 AAAAAAAAAAAAAAAAAAAAAA 0 A_Lower
HCAN B 1 A_Lower
Loop
Select:
TNT1 AAAAAAAAAAAAAAAAAAAAAA 0 A_Raise
HCAN B 1 A_Raise
Loop
Altfire:
Fire:
HCAN B 0 A_JumpIfNoAmmo("NoAmmo")
HCAN B 0 A_PlaySoundEx("weapon/rainflush","Weapon")
HCAN B 0 A_TakeInventory("HydroCannonAmmo",1)
HCAN KL 1 A_GiveInventory("HydroCannonShot_M1")
HCAN L 0 A_ChangeVelocity(-Cos(pitch)*2, 0, sin(pitch)*2,CVF_RELATIVE)
HCAN B 0 A_Refire
Goto Ready2
NoAmmo:
HCAN B 1 //ACS_Execute(979,0)
Goto Ready2
}
}

actor HydroCannonShot : BasicProjectile
{
var int user_damage;
Obituary "%o was hosed down by %k's Hydro Cannon."

+DONTBLAST
-NOGRAVITY
+FORCEXYBILLBOARD
Radius 20
Height 5
scale 4.5
damagetype "GizmoWater"
damage (3-user_damage)
speed 100
States
{
Spawn:
WTRB O 0
WTRB O 0 A_SpawnItemEx("HydroSplashFX",0,0,0,0,0,0,0,SXF_TRANSFERTRANSLATION)
HYDR BCD 4
WTRB O 0 A_SetScale(4.0)
HYDR B 0 A_SetUserVar("user_damage",1)
HYDR B 0 A_ScaleVelocity(0.6)
HYDR BCD 6
WTRB O 0 A_SetScale(3.5)
HYDR B 0 A_SetUserVar("user_damage",1)
HYDR B 0 A_ScaleVelocity(0.9)
SpawnFinal:
HYDR BCD 7
loop
Death:
WTRB K 0 A_Stop
WTRB K 0 A_ChangeFlag("NOGRAVITY",1)
WTRB K 0 A_SetScale(2.5)
WTRB KLMN 2
stop
}
}
