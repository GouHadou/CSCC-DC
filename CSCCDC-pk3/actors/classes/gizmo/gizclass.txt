actor Gizmo : CustomWepMan 
{
player.displayname "Gizmo" 
player.soundclass "Gizmo"
player.damagescreencolor "red"
player.forwardmove 0.8, 0.8
player.sidemove 0.78, 0.78
player.jumpz 13
player.startitem "HandCannon"
player.startitem "ClawGlovesWep"
player.startitem "BusterAmmo", 3
player.startitem "HandCannonAmmo", 3
player.startitem "FlameCannonAmmo", 14
player.startitem "LightningCannonAmmo", 14
player.startitem "FrostCannonAmmo", 14
player.startitem "GizmoCopy"

States
{
Spawn:
GIZM A 0
GIZM B 1
GIZM A 1
Goto Spawn+2
See:
GIZM BCDE 5 
Goto Spawn
Missile:
GIZM A 0 A_JumpIfInventory("ClawAttack",1,"CGlove")
GIZM A 1 
GIZM F 5
GIZM G 4
Goto Spawn
CGlove:
GIZM I 5
GIZM J 4 
GIZM A 2
goto Spawn
ClassDeath:
GIZM K 0 A_GiveInventory("ClassDeathPickup",1)
GIZM K 0 A_PlayerScream
GIZM K 5 ThrustThingZ(0,75,0,0)
GIZM KKKKKKKKKK 1 A_CheckFloor("DeathFinal")
GIZM K 5 A_ScaleVelocity(0.25)
GIZM K 0 ThrustThingZ(0, 10, 1, 1)
GIZM KKKKKKKKKK 1 A_CheckFloor("DeathFinal")
GIZM KKKKKKKKKK 1 A_CheckFloor("DeathFinal")
GIZM KKKKKKKKKK 1 A_CheckFloor("DeathFinal")
Goto DeathFinal
DeathFinal:
PLAY Z 0 //A_JumpIfInventory("TsukiHit", 1, "TsukiSoulDrop")
PLAY Z 1 A_CheckPlayerDone
wait
}
}
