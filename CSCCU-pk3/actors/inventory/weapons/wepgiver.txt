//_Wep : CMM or whoever's original weapon.
//_WepGiver : Replaces the Weps placed on the map so only those that can copy weapons and don't have the weapon can grab them.
//_WepDropped : The weapons the players drop; can refill ammo.

actor WepGiver : CustomInventory
{
inventory.pickupmessage "Power up! Mega Buster!"
inventory.pickupsound "weapon/weaponup"
inventory.amount 1
inventory.maxamount 1
inventory.respawntics 5
scale 2.0
+INVENTORY.AUTOACTIVATE
States
{
Spawn:
"----" A 0
"----" A 0 //A_SetTranslucent(1.0, 0)
"----" A 35 ACS_NamedExecuteAlways("CSACS_ITEMSET", 0)
loop
OtherSpawn:
Goto ColdFusionSpawn
MegaSpawn:
CopySpawn:
"----" A 0 A_SetTranslucent(1.0, 0)
"----" A 35 
goto Spawn
ColdFusionSpawn:
IceSparkySpawn:
MessatsuSpawn:
UkiyamaSpawn:
KnuxSpawn:
YoshiSpawn:
RozarkSpawn:
AdaraSpawn:
KorbySpawn:
KingYamatoSpawn:
FiniteZeroSpawn:
JackCorvusSpawn:
OldYDSpawn:
YellowDevilSpawn:
ChimeraManSpawn:
CutManMikeSpawn:
BreveSpawn:
Beed28Spawn:
BlazeSpawn:
BlazeOldSpawn:
TsukiSpawn:
TsukiOldSpawn:
TheBladeRodenSpawn:
M712Spawn:
BikdarkSpawn:
LegoSpawn:
SaviorSwordSpawn:
DaverisSpawn:
HumanDestroyerSpawn:
StonefunkSpawn:
RedEyesSpawn:
GizmoSpawn:
ZardSpawn:
SmashSpawn:
HilmanSpawn:
"----" A 35 A_SetTranslucent(0.5, 0)
goto Spawn
Pickup:
DUST B 0 A_JumpIfInventory("MegaBuster", 1, "WFail")
DUST B 0 A_JumpIfInventory("CanCopyWeapons", 1, "Mega")
DUST B 0 A_JumpIfInventory("LaggyHUD", 1, "CorpPickup")

fail
WFail:
DUST B 0
fail
Mega:
DUST B 0 A_GiveInventory("MegaBuster",1)
stop
CorpPickup:
//Give Frag Grenade unless stated otherwise
DUST B 0 A_JumpIfInventory("CorpFragGrenade", 1, "WFail")
TNT0 A 1 ACS_NamedExecuteAlways("CorpWeaponPickup",0,1,0)
goto WFail
Death:
"----" A 1 A_Fadeout
loop
}
}
