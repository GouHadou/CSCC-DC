actor ProtoGiver : ItemGiver replaces ProtoUpgrade
{
inventory.pickupmessage "Support Item! Proto Upgrade!"
States
{
Spawn:
WEA2 Q 0
Goto "ItemGiver::Spawn+1"
LaggyBlazkoSpawn:
COW1 A 0 A_SetTranslucent(1.0, 0)
COW1 A 35
goto Spawn
HilmanSpawn:
HDEF X 0 A_SetTranslucent(1.0, 0)
HDEF X 35
goto Spawn
SmashSpawn:
USCU F 0 A_SetTranslucent(1.0, 0)
USCU F 35
goto Spawn
Pickup:
DUST B 0 A_JumpIfInventory("CanUseItems", 1, "Give")
DUST B 0 A_JumpIfInventory("LaggyHUD", 1, "CorpPickup")
DUST B 0 A_JumpIfInventory("StandardSpecial", 1, "SmashPU")
DUST B 0 A_JumpIfInventory("HilItem", 1, "Hil")
fail
Give:
WEAP Q 0 A_GiveInventory("ProtoUpgrade")
stop
CorpPickup:
//Grand Dad
DUST B 0 A_JumpIfInventory("CorpGrandDad", 1, "WFail")
TNT0 A 1 ACS_NamedExecuteAlways("CorpWeaponPickup",0,2,1)
stop
SmashPU:
DUST B 0 A_GiveInventory("UltraSmasherClawsUpgrade", 1)
Stop
Hil:
DUST B 0 A_GiveInventory("HilDefend", 1)
stop
}
}
