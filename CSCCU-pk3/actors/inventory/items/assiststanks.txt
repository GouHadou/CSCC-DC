actor ETankGiver : ItemGiver replaces ETank
{
inventory.pickupmessage "Support Item! E-Tank!"
States
{
Spawn:
EBAL F 0
EBAL F 0 A_JumpIf(tid == 999, 2)
EBAL F 0 Thing_ChangeTID(0,999)
EBAL F 0
Goto "ItemGiver::Spawn+1"
Megaspawn:
"----" A 0 A_SetTranslucent(1.0, 0)
Goto Animation
HilmanSpawn:
HMED X 0 A_SetTranslucent(1.0, 0)
HMED X 35
goto Spawn
ZardSpawn:
RTEA A 0 A_SetTranslucent(1.0, 0)
RTEA A 35
goto Spawn
SmashSpawn:
HOTD A 0 A_SetTranslucent(1.0, 0)
HOTD A 35
goto Spawn

Animation:
EBAL EFEFEF 6
EBAL F 0 ACS_NamedExecuteAlways("CSACS_ITEMSET", 0)
loop
Pickup:
DUST B 0 A_JumpIfInventory("CanUseItems", 1, "Give")
DUST B 0 A_JumpIfInventory("IsZardFlag",1,"GiveRecoveryTea")
DUST B 0 A_JumpIfInventory("StandardSpecial", 1, "SmashPU")
DUST B 0 A_JumpIfInventory("StandardSpecialUSC", 1, "SmashPU")
DUST B 0 A_JumpIfInventory("HilItem",1,"Hil")
fail
Give:
EBAL E 0 A_GiveInventory("ETank", 1)
stop
GiveRecoveryTea:
DUST B 0 A_GiveInventory("RecoveryTea", 1)
Stop
SmashPU:
DUST B 0 A_GiveInventory("HotDog", 1)
Stop
Hil:
DUST B 0 A_GiveInventory("HilMediPack", 1)
stop
}
}
