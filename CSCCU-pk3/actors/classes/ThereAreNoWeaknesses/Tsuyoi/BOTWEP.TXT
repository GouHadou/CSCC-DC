actor StormPunchBotWep : Base8BDMWeapon
{

Loop

Loop

TSUY B 50 
TSUY B 0 A_TakeInventory("PunchFlag",5)
TSUY B 1 A_Refire
Goto Ready+1

TSUY B 0 A_FireCustomMissile("StormPunch2",0,0,8,0)
TSUY CD 3

TSUY CD 3

TSUY B 20

TSUY B 0 A_FireCustomMissile("StormPunch2",0,0,8,0)
TSUY CD 3
