// WEAPON EXAMPLE 2
//
// This weapon is called "Awsum Weapon". It is a throwing weapon so it uses a different animation.
// The weapon itself will throw 3 explosives in an upward arc.
//
// There are less comments here. See weapon example 1 for a full run down of everything.

actor RingSlingerWep : BaseCSCCWep 18001
{
Weapon.AmmoUse 0
Weapon.AmmoGive 28
Weapon.SlotNumber 5
Obituary "%o was scattered by %k's Throwing Rings."
Inventory.Pickupmessage "Power up! Ring Slinger!"
weapon.ammotype "RingAmmo"
inventory.pickupsound "weapon/weaponup"
+WEAPON.AMMO_OPTIONAL
inventory.icon "SLINGI"
Scale 2.0
States
{
Spawn:
CLRN A 1
loop
Ready:
CLRN A 0 ACS_ExecuteAlways(998,0,201) 
CLRN A 1 A_WeaponReady
Goto Ready+1
Deselect:
CLRN A 2 A_TakeInventory("ClawAttack",1)
TNT1 AAAAAAAAAAAAAAAAAAAAAA 0 A_Lower
CLRN A 1 A_Lower
Loop
Select:
CLRN A 0 A_GiveInventory("ClawAttack",1)
TNT1 AAAAAAAAAAAAAAAAAAAAAA 0 A_Raise
CLRN A 1 A_Raise
Loop
Fire:
CLRN A 0 A_JumpIfNoAmmo("NoAmmo")
CLRN A 0 A_JumpIfInventory("RingCount",0,"NoAmmo")
CLRN B 0 A_PlaySoundEx("misc/Ringthrow","Weapon")
CLRN B 0 A_TakeInventory("RingCount",1)
CLRN B 0 A_FireCustomMissile("RedRing",0,1,0,0)
CLRN BCDEFG 1
CLRN A 0 A_Refire
Goto Ready+1
hold:
CLRN A 1
CLRN A 0 A_Refire
Goto Fire
NoAmmo:
CLRN A 1 ACS_Execute(979,0)
Goto Ready+1
}
}

actor RedRing : ProjectileBase
{
PROJECTILE
+LOOKALLAROUND
+SKYEXPLODE
+DONTBLAST
+PIERCEARMOR
Radius 10
Height 5
scale 1.5
damagetype "RingBoomerang"
damage (10)
speed 60
States
{
Spawn:
SLNG ABCD 1
loop
Death:
RNFX ABCD 1 A_PlaySound("misc/Ring2")
RNFX A 0 A_SpawnItemEx("GoldRing",0,0,0)
RNFX ABCD 0 

Stop
}
}

actor RingCount : Ammo
{
inventory.amount 1
inventory.maxamount 999
}

actor RingAmmo : Ammo
{
inventory.amount 1
inventory.maxamount 28
}
