actor HaxMonitor : Weapon
{
Inventory.PickupMessage "HAAAAAAAAAAAAAAAAAAAAAAX!"
Obituary "%o just got pwn3d by %k's Flying Monitor"
Weapon.SlotNumber 1
+WEAPON.DONTBOB
+WEAPON.NOAUTOAIM
+INVENTORY.UNDROPPABLE
+WEAPON.CHEATNOTWEAPON
States
{
Spawn:
TNT1 A 0
stop
Ready:
DHAX E 1 A_WeaponReady(WRF_NOSWITCH)
loop
Deselect:
TNT1 AAAAAAAAAAAAAAAAAAAAAA 0 A_Lower
DHAX E 1 A_Lower
loop
Select:
TNT1 AAAAAAAAAAAAAAAAAAAAAA 0 A_Lower
DHAX E 1 A_Raise
loop
Fire:
DHAX E 0 A_JumpIfInventory("ScoutFlag",1,"OhShi")
DHAX E 0 A_TakeInventory("ScoutFlag",1)
DHAX F 0 A_PlaySoundEx("weapon/thehax","weapon")
DHAX FGFGFG 1
DHAX F 0 A_FireCustomMissile("Pwner",0,0)
DHAX FGFGFGFGFGFGFGFGFGFGFGFGFGFGFGFGFGFGFGFGFGFGFGFGFG 1
goto Ready
OhShi:
DHAX E 0 A_TakeInventory("ScoutFlag",1)
DHAX F 0 A_PlaySoundEx("weapon/thehax","weapon")
DHAX F 0 A_FireCustomMissile("MegaPwner",0,0)
DHAX FGFGFGFGFGFGFGFGFGFGFGFGFGFGFGFGFGFGFGFGFGFGFGFGFGFGFGFGFGFG 1
goto Ready
AltFire:
DHAX E 0 A_JumpIfInventory("ScoutFlag",1,"Ready")
DHAX E 0 A_PlaySoundEx("weapon/coilstart","Weapon")
DHAX FGFGFGFGFGFGFGFGFGFGFGFGF 1 A_GiveInventory("ScoutFlag",1)
goto Ready
}
}

{
inventory.amount 1
inventory.maxamount 1
}
