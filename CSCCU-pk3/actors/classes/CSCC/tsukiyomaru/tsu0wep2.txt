actor TsukiClassCosmicWeapon : Weapon
{
Weapon.SlotNumber 1
scale 2.0
Inventory.Pickupmessage "Power Up! I don't give two shits!"
+WEAPON.AMMO_OPTIONAL
+WEAPON.DONTBOB
+WEAPON.CHEATNOTWEAPON
+INVENTORY.UNDROPPABLE
States
{
Spawn:
TNT1 A 0 
stop
Ready:
TSU0 C 0 ACS_ExecuteAlways(998,0,416)
TSU0 C 0 A_GunFlash
goto Ready2
Ready2:
TSU0 C 1 A_WeaponReady(WRF_NOSWITCH)
loop
Deselect:
TNT1 AAAAAAAAAAAAAAAAAAAAAA 0 A_Lower
TNT1 C 1 A_Lower
Loop
TrueDeselect:
TNT1 AAAAAAAAAAAAAAAAAAAAAA 0 A_Lower
TNT1 C 1 A_Lower
Loop
Select:
TNT1 AAAAAAAAAAAAAAAAAAAAAA 0 A_Raise
TSU0 c 1 A_Raise
Loop
Fire:
TSU0 C 0 A_PlaySoundEx("weapon/napalm","Weapon")
TSU0 C 0 A_FireCustomMissile("CosmicPulse1",0,0)
TSU0 DED 4
TSU0 C 8
goto Ready2
AltFire:
TSU0 D 0 A_JumpIfInventory("TsuCosmicCounter",1,"LightTackle") 
TSU0 D 1 
goto Ready2
LightTackle:
TSU0 E 1 A_TakeInventory("TsuCosmicCounter",1)
TSU0 E 0 A_ChangeVelocity(25,0,sin(-pitch)*25,CVF_RELATIVE|CVF_REPLACE)
TSU0 E 0 A_FireCustomMissile("LightTackle",0,1,0,0)
TSU0 E 0 A_FireCustomMissile("LightTackleFX",0,1,0,0)
goto Ready2
Flash:
TNT1 A 1
stop
}
}
