actor BubbleArrowWep : Base8BDMWeapon
{
Weapon.AmmoUse 2
Weapon.AmmoGive 28
Weapon.SlotNumber 5
scale 2.0
Obituary "%o was cleaned up by %k's Bubble Arrow."
Inventory.Pickupmessage "Power up! Bubble Arrow!"
weapon.ammotype "BubbleArrowAmmo"
+WEAPON.AMMO_OPTIONAL
+WEAPON.DONTBOB
inventory.icon "BubArrow"
inventory.pickupsound "weapon/weaponup"
States
{
Spawn:
BARW K 1
loop
Ready:
BUBA B 0 ACS_ExecuteAlways(998,0,WCCS_KNUX)
BUBA B 1 A_WeaponReady
Goto Ready+1
Deselect:
TNT1 AAAAAAAAAAAAAAAAAAAAAA 0 A_Lower
BUBA B 1 A_Lower
Loop
Select:
TNT1 AAAAAAAAAAAAAAAAAAAAAA 0 A_Raise
BUBA B 1 A_Raise
Loop
Fire:
BUBA B 0 A_JumpIfNoAmmo("NoAmmo")
BUBA B 0 A_PlaySoundEx("weapon/mbuster","Weapon")
BUBA B 0 A_FireCustomMissile("BubbleArrowShot",0,1,8,0)
BUBA CD 3
BUBA B 25
BUBA B 0 A_Refire
goto Ready+1
NoAmmo:
BUBA B 1 ACS_Execute(979,0)
Goto Ready+1
Flash:
TNT1 A 1
stop
NoFlash:
TNT1 A 1
stop
}
}
