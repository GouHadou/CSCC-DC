actor NashGore_FlyingBlood : ClientObjects
{
 game Doom
 scale 2.5
 health 1
 radius 8
 height 1
 mass 1
 renderstyle translucent
 alpha 1.0
 +CORPSE
 
 +NOTELEPORT
 +NOBLOCKMAP
 states
 {
 Spawn:
  TNT1 A 0 A_CheckSight("Cheap")
  FBLD ABCD 1 A_SpawnItem("NashGore_FlyingBloodTrail",0,0,0,1)
  FBLD ABCD 1 A_JumpIf(waterlevel > 1, "Remove")
//  FBLD A 0 A_Fadeout(0.1)
  loop
 Cheap:
  FBLD ABCD 8 A_SpawnItem("NashGore_FlyingBloodTrail",0,0,0,1)
  FBLD ABCD 1 A_JumpIf(waterlevel > 1, "Remove")
//  FBLD A 0 A_Fadeout(0.1)
  loop
 Crash:
  TNT1 A 1 A_SpawnItem("NashGore_BloodSpot",0,0,0,1)
  stop
   Remove: // Anti-Lag - Removes blood in water.
      NULL A 0 
        stop
 }
}
