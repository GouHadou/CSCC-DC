actor ShurikenBuster : BaseMM8BDMWep
{
Weapon.AmmoUse 0
Weapon.AmmoGive 3

Obituary "%o was ninja'd by %k's shurikens."

+WEAPON.AMMO_OPTIONAL
+WEAPON.DONTBOB
+INVENTORY.UNDROPPABLE
+WEAPON.CHEATNOTWEAPON
inventory.icon "SHADI"
DropItem "ShadowBladeWep"
States
{
Spawn:
TNT1 A 1
stop
Ready:
ALBS A 0 ACS_ExecuteAlways(998,0,424)
ALBS A 20 A_WeaponReady
ALBS A 0 A_GiveInventory("RJumpAmmo", 1)
Goto Ready+1
Deselect:
ALBS A 0 A_SelectWeapon("RocketJumpWep")
TNT1 AAAAAAAAAAAAAAAAAAAAAA 0 A_Lower
ALBS A 1 A_Lower
Loop
Select:
TNT1 AAAAAAAAAAAAAAAAAAAAAA 0 A_Raise
ALBS A 1 A_Raise
Loop
Fire:
ALBS A 0 //SetPlayerProperty(0,1,0)
ALBS A 0 A_PlaySoundEx("weapon/icewaveloop","Weapon")
ALBS A 0 A_FireCustomMissile("SmallShuriken2",0,0,8,0,0,Random(-5,5))
ALBS BC 2
ALBS D 1
ALBS A 0 A_FireCustomMissile("SmallShuriken1",0,0,8,0,0,Random(-5,5))
ALBS BC 2
ALBS D 1
ALBS A 0 A_FireCustomMissile("SmallShuriken1",0,0,8,0,0,Random(-5,5))
ALBS BC 2
ALBS D 1
ALBS A 0 A_FireCustomMissile("SmallShuriken1",0,0,8,0,0,Random(-5,5))
ALBS BC 2
ALBS D 1
ALBS A 0 A_Refire
ALBS A 0 //SetPlayerProperty(0,0,0)
goto Ready+1
Altfire:
ALBS A 0 A_PlaySoundEX("weapon/shadow","Weapon")
ALBA A 0 A_FireCustomMissile("Shuriken2",0,0,8,-8,0,0)
ALBA A 0 A_FireCustomMissile("Shuriken1",-4,0,8,8,0,0)
ALBA A 0 A_FireCustomMissile("Shuriken1",4,0,8,8,0,0)
ALBS BC 2
ALBS D 1
ALBS A 15 
ALBS A 0 A_Refire
goto Ready+1
}
}
