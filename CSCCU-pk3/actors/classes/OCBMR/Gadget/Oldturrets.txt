//------[]
Flash:
TNT1 A 0 A_JumpIfInventory("2Soon2Switch",1,2)
TNT1 A 0 A_GiveInventory("GadgetAmmo",6)

loop
Reclaim:

TNT1 A 1 //ACS_Execute(979,0)
